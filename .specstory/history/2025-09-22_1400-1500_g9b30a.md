   492→        async deriveKey(passphrase) {
   493→            const encoder = new TextEncoder();
   494→            const data = encoder.encode(passphrase);
   495→            const hashBuffer = await crypto.subtle.digest('SHA-256', data);
   496→            return await crypto.subtle.importKey(
   497→                'raw',
   498→                hashBuffer,
   499→                { name: 'AES-GCM' },
   500→                false,
   501→                ['decrypt']
   502→            );
   503→        }

   505→        async decryptContent(encryptedData, key) {
   506→            const iv = this.base64ToArrayBuffer(encryptedData.iv);
   507→            const authTag = this.base64ToArrayBuffer(encryptedData.authTag);
   508→            const encrypted = this.base64ToArrayBuffer(encryptedData.encrypted);
   509→            
   510→            // Combine encrypted data with auth tag for AES-GCM
   511→            const combinedBuffer = new Uint8Array(encrypted.byteLength + authTag.byteLength);
   512→            combinedBuffer.set(new Uint8Array(encrypted), 0);
   513→            combinedBuffer.set(new Uint8Array(authTag), encrypted.byteLength);
   514→            
   515→            const decrypted = await crypto.subtle.decrypt(
   516→                {
   517→                    name: 'AES-GCM',
   518→                    iv: iv
   519→                },
   520→                key,
   521→                combinedBuffer
   522→            );
   523→            
   524→            return new TextDecoder().decode(decrypted);
   525→        }

   700→        convertMarkdownToHtml(markdown, decryptedContent = null) {

   705→            // First handle code blocks to protect them from other processing

   708→                const index = codeBlocks.length;

   710→                return `__CODE_BLOCK_${index}__`;
   711→            });

   714→            htmlContent = htmlContent

   716→                    const id = this.generateHeadingId(title);
   717→                    return `<h1 id="${id}">${title}</h1>`;
   718→                })
   719→                .replace(/^## (.+)$/gm, (match, title) => {
   720→                    const id = this.generateHeadingId(title);
   721→                    return `<h2 id="${id}">${title}</h2>`;
   722→                })
   723→                .replace(/^### (.+)$/gm, (match, title) => {
   724→                    const id = this.generateHeadingId(title);
   725→                    return `<h3 id="${id}">${title}</h3>`;
   726→                })

   728→                    const id = this.generateHeadingId(title);
   729→                    return `<h4 id="${id}">${title}</h4>`;
   730→                })
   731→                .replace(/^##### (.+)$/gm, (match, title) => {
   732→                    const id = this.generateHeadingId(title);
   733→                    return `<h5 id="${id}">${title}</h5>`;
   734→                });
   735→            
   736→            // Handle text formatting
   737→            htmlContent = htmlContent
   738→                .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
   739→                .replace(/\*([^*]+?)\*/g, '<em>$1</em>')
   740→                .replace(/`([^`]+?)`/g, '<code>$1</code>');
   741→            
   742→            // Handle images and links (images first to avoid conflicts)
   743→            htmlContent = htmlContent
   744→                .replace(/!\[([^\]]*)\]\(([^)]+)\)/g, '<img src="$2" alt="$1" style="max-width: 100%; height: auto;">')
   745→                .replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2">$1</a>');
   746→            
   747→            // Handle lists - improved handling
   748→            htmlContent = htmlContent.replace(/^(\s*)-\s+(.+)$/gm, (match, indent, content) => {
   749→                const level = Math.floor(indent.length / 2);
   750→                return `<li data-level="${level}">${content}</li>`;
   751→            });
   752→            
   753→            // Wrap consecutive list items in ul tags
   754→            htmlContent = htmlContent.replace(/(<li[^>]*>.*?<\/li>\s*)+/gs, (match) => {
   755→                return `<ul>${match}</ul>`;
   756→            });
   757→            
   758→            // Handle numbered lists
   759→            htmlContent = htmlContent.replace(/^(\s*)\d+\.\s+(.+)$/gm, (match, indent, content) => {
   760→                const level = Math.floor(indent.length / 2);
   761→                return `<li data-level="${level}">${content}</li>`;
   762→            });
   763→            
   764→            // Convert paragraphs - be more careful about existing HTML
   765→            const lines = htmlContent.split('\n');
   766→            let inList = false;
   767→            let result = [];

   769→            for (let i = 0; i < lines.length; i++) {

   773→                    result.push('');

   775→                }

   779→                    result.push(line);
   780→                    inList = line.includes('<li') || line.includes('<ul') || line.includes('<ol');
   781→                } else if (!inList) {
   782→                    // Only wrap in paragraph if not already HTML and not in a list
   783→                    result.push(`<p>${line}</p>`);
   784→                } else {
   785→                    result.push(line);
   786→                }
   787→            }
   788→            
   789→            htmlContent = result.join('\n');
   790→            
   791→            // Restore code blocks
   792→            codeBlocks.forEach((block, index) => {
   793→                htmlContent = htmlContent.replace(`__CODE_BLOCK_${index}__`, block);
   794→            });
   795→            
   796→            // Clean up empty paragraphs and fix common issues
   797→            htmlContent = htmlContent
   798→                .replace(/<p><\/p>/g, '')
   799→                .replace(/<p>(<[^>]+>)/g, '$1')  // Remove p tags around block elements
   800→                .replace(/(<\/[^>]+>)<\/p>/g, '$1');  // Remove closing p tags after block elements
   801→            
   802→            return htmlContent;
   803→        }
